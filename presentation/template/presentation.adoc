class CalculationServiceAdapter(
    private val calculationServiceRestClient: CalculationServiceRestClient,
    private val calculationServiceOperationSpec: CalculationServiceOperationSpecification
) : CalculationServicePort {

    override fun calculate(calculation: Calculation): CalculationResult {
        // these methods are defined in the adapter as kotlin extension functions, not in the domain
        val request = calculation.mapDomainToRequest(calculationServiceOperationSpec.requestMappingSpec())
        val path = calculation.getEndpointPath(calculationServiceOperationSpec.pathMappingSpec())

        val response = calculationServiceRestClient.request(path, request)

        return response.mapResponseToDomain(calculationServiceOperationSpec.responseMappingSpec())
    }
}
